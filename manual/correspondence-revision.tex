\chapter{File versions}

The correspondence module comes as a package with many different files and each of them
has their own version tag, you can get this information from the information block at the
top of the file, \MKIV\ users  get also a overview of the used files with the version at
the end of the log file.

\start

\setuptabulate[header=repeat]

\starttabulatehead
\NC \bf File \NC \bf Revision \NC \bf Description            \NC\NR
\stoptabulatehead

\startmodulecode

local pairs, context = global.pairs, global.context

context.starttabulate( { "|l|l|l|" } )
for k,v in pairs(correspondence.files) do
    context.NC()
    context("%s",correspondence.files[k]["name"])
    context.NC()
    context("%s",correspondence.files[k]["version"])
    context.NC()
    context("%s",correspondence.files[k]["description"])
    context.NC()
    context.NR()
end
context.stoptabulate()

\stopmodulecode

\stop

\stopcomponent
